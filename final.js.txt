function Cargar(){
    var id = parseInt(this.parentNode.parentNode.parentNode.id);
    var cargarNombreProducto = this.parentNode.firstElementChild.firstElementChild.firstElementChild.innerHTML;
    var cargarPrecioProducto = this.parentNode.firstElementChild.lastElementChild.lastElementChild.innerHTML;
    var cargarImgProducto = this.parentNode.parentNode.firstElementChild;
    
    id = id + Math.random();
    
    var objetoCarrito = {id,
                        cargarImgProducto,
                        cargarNombreProducto,
                        cargarPrecioProducto,
                        }; 


    aCarrito.push(objetoCarrito);
    contadorDeProductos++;
    spanNegro.innerHTML = contadorDeProductos;
    spanBlanco.innerHTML = parseInt(spanBlanco.innerHTML) + parseInt(cargarPrecioProducto);
}
